60  /*! \brief A relation between keys that is true if some conditions are true.
61  *
62  * A rule is composed of two parts, a left-hand side and some
63  * conditions. The left-hand side proposes a relation between some
64  * variables. The conditions represent the relations that are
65  * adequate to prove that the left-hand side relation is true.
66  *
67  * Note that the conditions are adequate to prove the relation,
68  * but not always necessary. There may be multiple rules with the
69  * same name but different conditions, and any of these rules can
70  * also prove the relation.
71  */
72  struct rule
73  {
74  /*! Name of the rule. */
75  std::string name;
76  /*! Number of variables on the left-hand side of the rule. */
77  uint lfs;
78  /*! Conditions that are adequate to prove the relation on the
79  * left-hand side of the rule.
80  */
81  std::vector<condition> conditions;
82 
83  MSGPACK_DEFINE_MAP(name, lfs, conditions);
84  };
