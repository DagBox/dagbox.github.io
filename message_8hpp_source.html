102  template <class iterator>
103  auto read_part(iterator & iter, iterator & end) -> part {
104  if (iter == end) {
105  throw exception::malformed("Expected message part is missing");
106  }
107  return std::move(*(iter++));
108  }
109 
110  template <class iterator>
111  auto read_optional(iterator & iter, iterator & end) -> optional_part {
112  // If there are no parts left, no optional part
113  if (iter == end) {
114  return boost::none;
115  }
116  // If the first part is empty, no optional part
117  part first = std::move(*iter);
118  if (first.size() == 0) {
119  return boost::none;
120  }
121  // If the first part is not empty, and there are no parts
122  // left, then we found the optional part
123  ++iter;
124  if (iter == end) {
125  return std::move(first);
126  }
127  // If the first part is not empty and the second part is, then
128  // we found the optional part
129  part second = std::move(*iter);
130  if (second.size() == 0) {
131  return std::move(first);
132  }
133 
134  // If both the first and the second part are not empty, then
135  // we weren't sent an optional part at all
136  throw exception::malformed("Expected optional message part "
137  "is malformed");
138  }
139 
140  template <class iterator>
141  auto read_many(iterator & iter, iterator & end) -> many_parts {
142  many_parts parts;
143  while (iter != end && (*iter).size() != 0) {
144  parts.push_back(std::move(*(iter++)));
145  }
146  return parts;
147  }
